    <script>
        const SCRIPT_URL = "ضع_رابط_جوجل_هنا";
        const bc = new BroadcastChannel('wheel_control');

        function login() {
            if(document.getElementById('pass').value === 'admin-sunnan1212') {
                document.getElementById('login-panel').style.display = 'none';
                document.getElementById('admin-panel').style.display = 'block';
                loadData();
            } else { alert("كلمة المرور خطأ"); }
        }

        function loadData() {
            fetch(`${SCRIPT_URL}?action=getQualified`)
                .then(res => res.json())
                .then(names => {
                    document.getElementById('stats').innerText = `عدد المتأهلين (إجابات صحيحة وبدون تكرار): ${names.length}`;
                });
        }

        function triggerSpin() {
            // كود اختيار فائز عشوائي وإرساله لصفحة العرض
            bc.postMessage({ action: 'spin', winner: 'فائز عشوائي', prize: 'جائزة عشوائية' });
        }
    </script>
